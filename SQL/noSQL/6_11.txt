db.getCollection('customer').aggregate([
  {
    $lookup: {
      from: "order",
      localField: "customer_id",
      foreignField: "customer_id",
      as: "orders"
    }
  },
  {
    $match: {
      orders: { $eq: [] }
    }
  },
  {
    $project: {
      _id: 0,
      customer_id: 1,
      customer_name: 1
    }
  }
])
